Backend Engine

API Reference

Express.js REST API connected to MongoDB Atlas. All routes return JSON.

Foundation
server.js 
require('dotenv').config();
const express  = require('express');
const mongoose = require('mongoose');
const cors     = require('cors');
const app      = express();

// Middleware
app.use(cors());
app.use(express.json());

// MongoDB Atlas
mongoose.connect(process.env.MONGO_URI)
  .then(() => console.log('‚úî Atlas connected'));

// Schema
const Post = mongoose.model('Post', new mongoose.Schema({
  title:     { type: String,  required: true },
  category:  { type: String,  default:  'general' },
  content:   { type: String,  required: true },
  author:    { type: String,  default:  'Anonymous' },
  excerpt:   { type: String },
  createdAt: { type: Date,    default:  Date.now },
}));

module.exports = { app, Post };

POST /api/posts Create a new blog entry
Handler
// POST /api/posts
app.post('/api/posts', async (req, res) => {
  const { title, category,
          content, author, excerpt } = req.body;
  const post = new Post({
    title, category, content,
    author, excerpt
  });
  await post.save();
  res.status(201).json(post);
});

PATCH /api/posts/:id Update existing post content
Handler
// PATCH /api/posts/:id
app.patch('/api/posts/:id',
  async (req, res) => {
  const post = await Post
    .findByIdAndUpdate(
      req.params.id,
      { $set: req.body },
      { new: true }
    );
  if (!post) return res.status(404)
    .json({ error: 'Not found' });
  res.json(post);
});
Notes

DELETE /api/posts/:id Remove a blog post permanently
Handler
// DELETE /api/posts/:id
app.delete('/api/posts/:id',
  async (req, res) => {
  const post = await Post
    .findByIdAndDelete(req.params.id);
  if (!post) return res.status(404)
    .json({ error: 'Not found' });
  res.json({ deleted: post._id });
});

Step 2 ‚Äî Vercel Configuration
‚ñ≤

Configure vercel.json for Serverless Functions

This tells Vercel to route all API calls to your Express app as a serverless function.

vercel.json
{
  "version": 2,
  "builds": [
    {
      "src": "server.js",
      "use": "@vercel/node"
    }
  ],
  "routes": [
    {
      "src":  "/api/(.*)",
      "dest": "server.js"
    },
    {
      "src":  "/(.*)?",
      "dest": "public/index.html"
    }
  ]
}
